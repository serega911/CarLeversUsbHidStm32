typedef struct {
	uint8_t report_id;
	uint8_t modifier;
	uint8_t reserved;
	uint8_t keycode_data[5];
}__attribute__((packed, aligned (1))) KeyboardReport_t;


extern USBD_HandleTypeDef hUsbDeviceFS;
static KeyboardReport_t keyboardReport = {0};


keyboardReport.report_id = 0x01;

BSP_LED_Toggle(LED_GREEN);
keyboardReport.modifier = 0x02;
keyboardReport.keycode_data[0] = 0x10;
USBD_HID_SendReport(&hUsbDeviceFS, (uint8_t *)&keyboardReport, sizeof(keyboardReport));
osDelay(100);

keyboardReport.modifier = 0x00;
keyboardReport.keycode_data[0] = 0x00;
USBD_HID_SendReport(&hUsbDeviceFS, (uint8_t *)&keyboardReport, sizeof(keyboardReport));
osDelay(1000);
